参数为回调函数

修改于: 2024-11-20

dart的定义跟ts有点不一样, 比如在ts那边是

onChange: (arg: String) => void

在 dart 中是

Function(String) onChange;

举例, 实现一个 SearchBox

import 'package:flutter/material.dart';

import '../global_colors.dart';

class SearchBox extends StatefulWidget{

  final Function(String) onChange;

  const SearchBox({super.key, required this.onChange});

  @override
  State<SearchBox> createState() => _SearchBoxState();
}

class _SearchBoxState extends State<SearchBox> {

  final TextEditingController _controller = TextEditingController();
  String _searchText = '';

  void _onSearch(String value) {
    setState(() {
      _searchText = value; // 更新搜索文本
    });
    // 在这里可以添加实时搜索逻辑
    debugPrint('Searching for: $_searchText');

    // 触发回调
    widget.onChange(value);
  }

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(8.0),
      child: Column(
        children: [
          Row(
            children: [
              Expanded(
                child: ConstrainedBox(
                  constraints: BoxConstraints(
                      maxHeight: 25,
                      maxWidth: 200,
                  ),
                  child: TextField(
                    controller: _controller,
                    onChanged: _onSearch, // 处理输入变化
                    style: TextStyle(
                      // color: Theme.of(context).colorScheme.surface,
                      fontSize: 14.0,
                    ),
                    decoration: InputDecoration(
                      contentPadding: const EdgeInsets.symmetric(vertical: 6.0, horizontal: 12),
                      // hintText: '请输入搜索内容',
                      prefixIcon: Icon(Icons.search),
                      // contentPadding: EdgeInsets.all(10),
                      border: OutlineInputBorder(
                          borderRadius: BorderRadius.circular(10),
                          borderSide: BorderSide.none
                      ),
                      filled: true,
                      isCollapsed: true,
                      fillColor: GlobalColors.leftSideBarBgColor,
                    ),
                  ),
                ),

              ),
            ],
          ),
          // SizedBox(height: 20),
          // if (_searchText.isNotEmpty)
          //   Text(
          //     'Search Results for: $_searchText',
          //     style: TextStyle(fontSize: 18),
          //   ),
        ],
      ),
    );
  }

}
