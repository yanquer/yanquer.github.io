音频播放库

安装

cargo add rodio

使用, 一个封装的单例音频播放器

use rodio::{source::Source, Decoder, OutputStream, Sink};
use std::fs::File;
use std::io::BufReader;
use std::sync::{Arc, Mutex};
use std::time::Duration;
use lazy_static::lazy_static;
use log::info;

pub struct AudioPlayer {
    sink: Sink,
    _stream: OutputStream, // 需要持有_stream，否则会被释放
}

impl AudioPlayer {
    // 获取单例实例
    pub fn get_instance() -> Arc<Mutex<AudioPlayer>> {
        // 使用 Arc 和 Mutex 来实现线程安全的单例
        static mut INSTANCE: Option<Arc<Mutex<AudioPlayer>>> = None;
        unsafe {
            INSTANCE
                .get_or_insert_with(|| {
                    Arc::new(Mutex::new(AudioPlayer::new()))
                })
                .clone()
        }
    }

    // 创建一个新的AudioPlayer实例
    pub fn new() -> Self {
        let (_stream, stream_handle) = OutputStream::try_default().unwrap();
        let sink = Sink::try_new(&stream_handle).unwrap();
        AudioPlayer { sink, _stream }
    }

    // 播放音频文件
    pub fn play(&self, file_path: String) {
        let file = BufReader::new(File::open(&file_path).unwrap());
        let source = Decoder::new(file).unwrap();
        self.sink.append(source);
        self.sink.play();
        println!("Playing: {}", file_path);
        info!("Playing: {}", file_path);
    }

    // 暂停播放
    pub fn pause(&self) {
        self.sink.pause();
        println!("Paused.");
        info!("Paused.");
    }

    // 继续播放
    pub fn resume(&self) {
        self.sink.play();
        println!("Resumed.");
        info!("Resumed.");
    }

    // 停止播放
    pub fn stop(&self) {
        self.sink.stop();
        println!("Stopped.");
        info!("Stopped.");
    }

    // 跳转到指定时间（秒）
    pub fn seek(&self, file_path: String, seconds: u64) {
        self.sink.stop(); // 停止当前播放
        let file = BufReader::new(File::open(file_path).unwrap());
        let source = Decoder::new(file).unwrap().skip_duration(Duration::from_secs(seconds));
        self.sink.append(source);
        self.sink.play();
        println!("Seeked to {} seconds.", seconds);
        info!("Seeked to {} seconds.", seconds);
    }
}
