Webpack仅压缩CSS文件

修改于: 2025-02-06

此处先提一下 Webpack 的生成大致效果.

比如入口文件是 app.js, app.js 中导入了几个 css 文件 当CSS的加载器没有 style-loader 时，Webpack 会将所有的 css 文件打包成一个 app.css 文件(默认)，不会将其插入到 html 文件中:

module: {
  rules: [
      {
          test: /\.css$/,
          use: [
              MiniCssExtractPlugin.loader,
              'css-loader',
          ],
      },
  ],
},

而当使用了 style-loader 时，Webpack 会将所有 css 文件作为 style 插入到 html 文件中:

module: {
    rules: [
        {
            test: /\.css$/,
            use: [
              MiniCssExtractPlugin.loader,
                'style-loader',
                'css-loader',
            ],
        },
    ],
},

仅压缩CSS文件示例

webpack.config.js 文件

/**
* This file can be edited to customize webpack configuration.
* To reset delete this file and rerun theia build again.
*/
// @ts-check
const path = require("path");
const resolvePackagePath = require('resolve-package-path');
const CopyWebpackPlugin = require('copy-webpack-plugin');
const CompressionPlugin = require('compression-webpack-plugin')
const MiniCssExtractPlugin = require('mini-css-extract-plugin')
const CssMinimizerPlugin = require('css-minimizer-webpack-plugin');
const glob = require('glob');
const RemoveEmptyScriptsPlugin = require('webpack-remove-empty-scripts');
const {existsSync, mkdirSync} = require("node:fs");


// 使用 glob 自动获取所有 CSS 文件
const skinDarkPath = "./resources/tinymce/skins/ui/oxide-dark-2/"
const entriesTinymceCss = glob.sync(
    skinDarkPath + '*.css'
).reduce((acc, path_) => {
    const name = path.basename(path_);
    const entry = name.replace(".css", "")
    acc[entry] = path_;
    return acc;
}, {});

// 或者只定义需要编译的css
console.log("entriesTinymceCss: ", entriesTinymceCss)
const customCss = {
    "content": skinDarkPath + "content.css",
    "skin": skinDarkPath + "skin.css",
}
const tinymceExport = {
    mode: 'production', // 生产模式自动压缩 JS（但此处不需要）
    entry: customCss,
    output: {
        filename: 'tmp-js-[name].min.js', // 虽然只处理 CSS，但需要一个文件名, 后面通过 RemoveEmptyScriptsPlugin 移除
        path: path.resolve(__dirname, 'resources', 'tinymce', 'skins', 'ui', 'oxide-dark-2'),
        // clean: false, // 每次构建前是否清理输出目录
    },
    module: {
        rules: [
            {
                test: /\.css$/,
                use: [
                    MiniCssExtractPlugin.loader,
                    'css-loader',
                ],
            },
        ],
    },
    plugins: [
        new RemoveEmptyScriptsPlugin(), // 移除由 CSS 入口生成的空 JS 文件
        new MiniCssExtractPlugin({
            filename: '[name].min.css', // 输出的压缩 CSS 文件名
        }),
    ],
    // 压缩
    optimization: {
        minimizer: [
            `...`, // 保留默认的 JS 最小化器
            new CssMinimizerPlugin(),
        ],
        // 开发环境下也启用 CSS 优化
        minimize: true,
    },
};

module.exports = [
    tinymceExport,
];
说明

因为webpack必须有输入输出, 加上此处没有要用到的js, 所以此处直接将所有 css 文件映射作为入口;

输出随便指定什么, 再通过 RemoveEmptyScriptsPlugin 插件移除生成的空 js 文件即可;

  • MiniCssExtractPlugin 插件用于提取 css 文件;
  • CssMinimizerPlugin 用于压缩 css 文件;
