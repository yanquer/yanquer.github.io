用于将图片集合转换为pdf文件

安装:

pip install img2pdf

命令行使用

img2pdf img1.png img2.jpg -o out.pdf

使用封装

一个封装

import logging
import os

import img2pdf
from PIL import Image
from natsort import natsorted

_LOGGER = logging.getLogger(__name__)


class SortUtil(object):

    @staticmethod
    def sorted(data: []):
        """
        对列表自然排序
        :param data: 需要排序的列表
        :return:
        """
        return natsorted(data)


class PdfConverter(object):

    @staticmethod
    def check_image_size_right(image_file: str, min_size_inches=3.0, dpi=72) -> bool:
        """
        检查图片尺寸是否大于 min_size_inches 英寸
            检查是因为如果存在小于3英寸的图片, img2pdf 没法转换
        :param image_file: 图片路径
        :param min_size_inches: 最小尺寸（英寸）
        :param dpi: 分辨率（每英寸像素数）
        :return:
        """
        img = Image.open(image_file)
        min_size = min_size_inches * dpi
        return img.width >= min_size and img.height >= min_size

    @classmethod
    def convert_img_list_to_pdf(
            cls,
            input_files: [str],
            output_pdf: str,
    ) -> None:
        """
        将 图片列表 保存为 pdf
        :param input_files: 图片列表, 注意图片顺序
        :param output_pdf:  输出pdf文件
        :return: None
        """
        parent_path = os.path.dirname(output_pdf)
        if not os.path.exists(parent_path):
            os.makedirs(parent_path)

        for one in input_files:
            if not cls.check_image_size_right(one):
                _LOGGER.debug(f"图片 {one} 尺寸太小, 不将其放入pdf, 忽略此输出")
                input_files.remove(one)

        with open(output_pdf, 'wb') as f:
            f.write(img2pdf.convert(input_files))

    @classmethod
    def convert_image_dir_to_pdf(
            cls,
            input_dir: str,
            output_pdf: str,
    ) -> None:
        """
        将指定文件夹 内的所有图片 保存为 pdf文件
        :param input_dir:   含有图片的文件夹
        :param output_pdf:  输出pdf文件
        :return: None
        """
        images = SortUtil.sorted(t for x in os.listdir(input_dir) if os.path.isfile(t := os.path.join(input_dir, x)) and cls.is_image(t))
        return cls.convert_img_list_to_pdf(images, output_pdf)

    @staticmethod
    def is_image(file_: str) -> bool:
        """
        判读文件是否是图片文件
        :param file_: 需要判断的文件
        :return:
        """
        if not os.path.isfile(file_):
            return False
        file_ext = os.path.splitext(file_)[1]
        return file_ext.lower() in ('.jpg', '.jpeg', '.png')
