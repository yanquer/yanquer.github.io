函数可变参数处理库

API

常用

va_list vl;     // va_list指针, 用于va_start取可变参数，为char*

va_start(vl, num);       //取得可变参数列表中的第一个值

C 可变参数列表

原理就是, 参数是通过栈传递, 栈的空间在内存上是连续的, 所以只要按照顺序获取内存块的大小就好, 额外需要注意的是, 自行实现时需要考虑内存一致性(即内存块之间并不是完全连续, 可能存在间隙)

格式化字符串参数

根据格式化字符串来判断参数类型

实现一个 printf

#include <stdio.h>
#include<stdarg.h>

/*
*
*  %d: 整数
    %f: 浮点数
    %s: 字符串
    %c: 单个字符
* */
void min_printf(char *fmt, ...)
{
    va_list ap;
    char *p, *s_val;
    int i_val;
    double d_val;

    va_start(ap, fmt);
    for (p = fmt; *p; p++) {
        if(*p != '%') {
            putchar(*p);
            continue;
        }
        switch(*++p) {
            case 'd':
                i_val = va_arg(ap, int);
                printf("%d", i_val);
                break;
            case 'f':
                d_val = va_arg(ap, double);
                printf("%f", d_val);
                break;
            case 'c':
                i_val = va_arg(ap, int);
                putchar(i_val);
                break;
            case 's':
                for (s_val = va_arg(ap, char *); *s_val; s_val++)
                    putchar(*s_val);
                break;
            default:
                putchar(*p);
                break;
        }
        fflush(stdout);
    }
    va_end(ap);
}

int main(int argc, char *argv[]) {

    min_printf("痘 sadome asodo %s 00 %d 00 %c 00 %f", "-=-=-=-", 99, 'x', 9.9);
    exit(0);
}

输出

痘 sadome asodo -=-=-=- 00 99 00 x 00 9.900000

已知参数类型与个数的情况

由于已知参数类型, 直接转换即可

#include <stdio.h>
#include<stdarg.h>

/*
* num: 一共有多少个参数
* */
int num_count_printf(int num, ...){
    int i, result=0;
    va_list vl;                 //va_list指针，用于va_start取可变参数，为char*
    va_start(vl, num);          //取得可变参数列表中的第一个值
    // printf("num:%d, vl:%d\n", num, *vl);
    printf("num:%d, \n", num);

    // 这里num表示可变参数列表中有多少个参数
    for (i = 0; i < num; i++)
    {
        // vl往后跳过4个字节（sizeof(int)大小）指向下一个参数，返回的是当前参数（而非下一个参数）
        result += va_arg(vl, int);
        printf("result: %d, \n", result);
        // printf("in for  result:%d,  *vl:%d\n", result, *vl);// 这里打印下，可以看出，vl总是指向result后面的那个参数
    }
    // 结束标志
    va_end(vl);

    return result;
}

int main(int argc, char *argv[]) {
    printf("%d", num_count_printf(5, 1, 2, 3, 4, 5));
}

输出

result:1,
result:3,
result:6,
result:10,
result:15,
15

利用可变参数列表实现日志记录

注解

可变参数的参数传递, 只有 va_start 后传递 va_list

#include <stdio.h>
#include<stdarg.h>

/*
*
*  %d: 整数
    %f: 浮点数
    %s: 字符串
    %c: 单个字符
* */
void logger_printf_and_to_file(char *level, char *fmt, va_list ap)
{
    // va_list ap;
    char *p, *s_val;
    int i_val;
    double d_val;

    // 记录结果
    char msgs[1024] = "";
    int msgs_index = 0;

    // 临时缓存拼接 数字
    char tmp[256] = "";
    memset(tmp, 0, 256);
    char *t;

    // va_start(ap, fmt);
    for (p = fmt; *p; p++) {
        if(*p != '%') {
            msgs[msgs_index++] = *p;
            continue;
        }
        switch(*++p) {
            case 'd':
                i_val = va_arg(ap, int);
                sprintf(tmp, "%d", i_val);
                for ( t=tmp; *t; t++ )
                    msgs[msgs_index++] = *t;
                break;
            case 'f':
                d_val = va_arg(ap, double);
                sprintf(tmp, "%f", d_val);
                for ( t=tmp; *t; t++ )
                    msgs[msgs_index++] = *t;
                break;
            case 'c':
                i_val = va_arg(ap, int);
                msgs[msgs_index++] = (char)(i_val);
                break;
            case 's':
                for (s_val = va_arg(ap, char *); *s_val; s_val++)
                    msgs[msgs_index++] = *s_val;
                break;
            default:
                msgs[msgs_index++] = *p;
                break;
        }

    }
    va_end(ap);

    char ret[1024];
    sprintf(ret, "[%s] => %s", level, msgs);

    printf("%s \n", ret);
    fflush(stdout);
    log_to_file(ret, NULL);
}


void info(char *fmt, ...){
    va_list ap;
    va_start(ap, fmt);
    logger_printf_and_to_file("info", fmt, ap);
}

void warn(char *fmt, ...){
    va_list ap;
    va_start(ap, fmt);
    logger_printf_and_to_file("warn", fmt, ap);
}

void error(char *fmt, ...){
    va_list ap;
    va_start(ap, fmt);
    logger_printf_and_to_file("error", fmt, ap);
}

实现 path.join

#include <unistd.h>
#include <stdarg.h>

char* path_join(int num_args, ...) {
    va_list args;
    size_t total_length = 0;
    char *sep = get_sep(); // \\ 或者 /

    // 计算所有路径的总长度
    va_start(args, num_args);
    for (int i = 0; i < num_args; i++) {
        const char *path = va_arg(args, const char*);
        total_length += strlen(path) + strlen(sep); // +1 for '/' or '\0'
    }
    va_end(args);

    // 分配内存
    char *ret_path = (char *)malloc(total_length);
    if (ret_path == NULL) return NULL; // 内存分配失败
    memset(ret_path, 0, total_length);

    // 重新开始处理参数
    va_start(args, num_args);
    char *ptr = ret_path;

    for (int i = 0; i < num_args; i++) {
        const char *path = va_arg(args, const char*);

        // 如果不是第一个路径，添加分隔符
        if (i > 0) {
            // *ptr++ = '/';
            strcpy(ptr, sep);
            ptr += strlen(sep);
        }

        // 复制路径到结果中
        strcpy(ptr, path);
        ptr += strlen(path);
    }
    va_end(args);

    return ret_path;
}

注解

记得外部使用结束后回收 ret_path

free(ret_path);
