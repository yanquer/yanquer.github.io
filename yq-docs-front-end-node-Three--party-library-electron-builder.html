将应用构建成包

比较流行的使用方式是在项目目录下放置一个 electron-builder.yml 文件

# id标识
appId: com.yq.flower

# 产品名称, 一般用作打包名称
productName: Screen Flower

# 版权
copyright: Copyright © 2024 YanQue

# 目录配置
directories:
  # 输出位置
  output: dist
  buildResources: resources

# 打包文件配置
files:
  - from: .
    filter:
      - package.json
      - app
  # 排除 .vscode 目录下的所有文件
  - '!**/.vscode/*'
  # 排除 src 目录下的所有文件
  - '!src/*'
  # 排除配置文件
  - '!electron.vite.config.{js,ts,mjs,cjs}'
  - '!{.eslintignore,.eslintrc.cjs,.prettierignore,.prettierrc.yaml,dev-app-update.yml,CHANGELOG.md,README.rst}'
  # 排除环境变量文件和 npm 配置文件
  - '!{.env,.env.*,.npmrc,pnpm-lock.yaml}'

# 发布设置
publish: null

# 是否使用 asar 压缩
#asar: false

# 额外配置打包的资源文件
extraResources:
#  - resources
  - from: resources
    to: resources

## 指定从asar归档中解压的文件或文件夹
#asarUnpack:
#  # 解压 resources 目录下的所有内容
#  - resources/**

# 不需要签名公证
#afterSign: "electron-builder-notarize"
# mac下打包dmg配置
dmg: {
  icon: 'resources/icons/screen-flower.icns',
  background: 'resources/icons/bg.jpg',
  contents: [
    { # 这个是右边图标及内容
      x: 410,
      y: 190,
      type: 'link',
      path: '/Applications'
    },
    { # 你左边的图标
      x: 130,
      y: 190,
      type: 'file'
    }
  ],
  window: { # 这里是整个窗口的大小
    height: 380,
    width: 540
  }
}

mac: {
  # 应用程序图标
  icon: 'resources/icons/screen-flower.icns',
  # 应用程序包名
  artifactName: '${productName}-${platform}-${arch}-${version}.${ext}',
  target: [ # 要打的包的格式类型设置
            'dmg',
            'zip' # 这里注意更新的时候，mac只认zip格式的包
  ],
  # 是否进行苹果官方的签名
  notarize: false,

  # 安装后执行文件名称, 默认是 package.json 中的 name 字段 (windows下默认是 productName)
  # executableName: xxx
}
