获取时间戳

// 获取当前时间
time_t raw_time;
struct tm *time_info;
time(&raw_time);
time_info = localtime(&raw_time);

printf("[%d-%02d-%02d %02d:%02d:%02d]\n",
          time_info->tm_year + 1900, time_info->tm_mon + 1, time_info->tm_mday,
          time_info->tm_hour, time_info->tm_min, time_info->tm_sec)
