rpc-websockets

修改于: 2025-02-17
相关资源
  • npm地址: https://www.npmjs.com/package/rpc-websockets
  • github地址: https://github.com/elpheria/rpc-websockets

一个用于websocket的rpc库, 可以创建rpc服务/客户端.

不知道为什么这个库准备废弃了... (后面看了下, 说没有收到原作者的回复, 所以废弃v10, ???)

安装:

npm install rpc-websockets --save
# yarn add rpc-websockets

使用

var WebSocket = require('rpc-websockets').Client
var WebSocketServer = require('rpc-websockets').Server

// instantiate Server and start listening for requests
var server = new WebSocketServer({
  port: 8080,
  host: 'localhost'
})

// register an RPC method
server.register('sum', function(params) {
  return params[0] + params[1]
})

// ...and maybe a protected one also
server.register('account', function() {
  return ['confi1', 'confi2']
}).protected()

// create an event
server.event('feedUpdated')

// get events
console.log(server.eventList())

// emit an event to subscribers
server.emit('feedUpdated')

// close the server
server.close()

// instantiate Client and connect to an RPC server
var ws = new WebSocket('ws://localhost:8080')

ws.on('open', function() {
  // call an RPC method with parameters
  ws.call('sum', [5, 3]).then(function(result) {
    require('assert').equal(result, 8)
  })

  // send a notification to an RPC server
  ws.notify('openedNewsModule')

  // subscribe to receive an event
  ws.subscribe('feedUpdated')

  ws.on('feedUpdated', function() {
    updateLogic()
  })

  // unsubscribe from an event
  ws.unsubscribe('feedUpdated')

  // login your client to be able to use protected methods
  ws.login({'username': 'confi1', 'password':'foobar'}).then(function() {
    ws.call('account').then(function(result) {
      require('assert').equal(result, ['confi1', 'confi2'])
    })
  }).catch(function(error) {
    console.log('auth failed')
  })

  // close a websocket connection
  ws.close()
})

遇到过的问题

客户端已经报错的链接出现多次链接

因为默认是自动重连的, 需要关闭 reconnect:

const WebSocket = require('rpc-websockets').Client

ws = new WebSocket('ws://localhost:8980/ws', {
          autoconnect: true,
          // 如果是外部实现重试机制, 那么此处自动重试必须关掉
          reconnect: false
      })

官方api说明: https://github.com/elpheria/rpc-websockets/blob/HEAD/API.md#wsconnect
