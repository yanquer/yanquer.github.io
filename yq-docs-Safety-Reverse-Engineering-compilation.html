常见约定寄存器参数及作用

寄存器 作用  
%rax 传递返回值  
%rdi 传递第一个参数  
%rsi 传递第二个参数  
%rdx 传递第三个参数  
%rcx 传递第四个参数(不一定)  
%r8 传递第五个参数  
%r9 传递第六个参数 callee-owned
%rsp 栈顶指针 caller-owned
%rbx 临时变量 caller-owned
%rbp 栈基指 caller-owned
%r12-%15 临时变量 caller-owned
%rip 存储下一条要执行的指令  
%eflags flags和条件判断的结果标志位  
xmmO 用来传递第一个double 参数  
xmm1 用来传递第二个dioubie 參费  
     
SP 栈寄存器  
LR 保存函数返回地址  
PC 当前运行到的地址  
R0-R3 用来放函数的参数，执行完后，R0存放返回值  
R4-R7 用来放函数的局部变量  

注解

rdi是8字节的，4字节的时候对应的就是edi。

callee-owned:
如果caller要使用这些寄存机，那么它在调用callee前，要把这些寄存器保存好。
caller-owned:
如果callee要使用这些寄存器，那么它就要保存好这些寄存器的值，并且返回到caller的时候要将这些值恢复。

寄存器总览-x86

通用寄存器

rax - Register A Extended (64 位)

主要用于算术运算和存放返回值

  • eax - Extended A Register (32 位)
  • ax - A Register (16 位)
  • al - A Low Register (8 位)
rbx - Register B Extended (64 位)

可以作为基址寄存器，通常在内存访问中使用

  • ebx - Extended B Register (32 位)
  • bx - B Register (16 位)
  • bl - B Low Register (8 位)
rcx - Register C Extended (64 位)

常用于计数器，特别是在循环和字符串操作中。

  • ecx - Extended C Register (32 位)
  • cx - C Register (16 位)
  • cl - C Low Register (8 位)
rdx - Register D Extended (64 位)

通常用于存放乘法和除法运算的第二个操作数或结果。

  • edx - Extended D Register (32 位)
  • dx - D Register (16 位)
  • dl - D Low Register (8 位)
rsi - Register Source Index Extended (64 位)

源操作数索引寄存器，常用于字符串和数组操作 x86中存储第二个参数.

  • esi - Extended Source Index Register (32 位)
  • si - Source Index Register (16 位)
rdi - Register Destination Index Extended (64 位)

目标操作数索引寄存器，常用于字符串和数组操作。 x86中存储第一个参数.

  • edi - Extended Destination Index Register (32 位)
  • di - Destination Index Register (16 位)
rsp - Register Stack Pointer Extended (64 位)

指向当前栈顶，管理函数调用和局部变量。

  • esp - Extended Stack Pointer Register (32 位)
  • sp - Stack Pointer Register (16 位)
rbp - Register Base Pointer Extended (64 位)

用于管理函数调用栈帧，提供对局部变量和参数的访问。 (一般作当前函数的基址指针)

  • ebp - Extended Base Pointer Register (32 位)
  • bp - Base Pointer Register (16 位)

指令指针寄存器

rip - Register Instruction Pointer (64 位)
指向下一条要执行的指令。

程序状态寄存器

rflags - Register Flags (64 位)
存储 CPU 状态和控制标志，如进位标志、零标志等。

控制寄存器

(Control Register)

cr0 - 控制寄存器 0
控制保护模式、虚拟内存等。
cr1 - 控制寄存器 1
一般未使用。
cr2 - 控制寄存器 2
存放最近一次页错误的线性地址。
cr3 - 控制寄存器 3
存放当前页表的物理地址。
cr4 - 控制寄存器 4
控制额外的 CPU 功能，如分页等。

示例-x86解析

注解

以下均使用x86机器编译

交换a,b的值

源码(此处有个小bug, tmp只声明未分配地址, Windows上运行可能会地址访问错误)

void swap(int *a, int *b){
  int *tmp;
  *tmp = *a;
  *a = *b;
  *b = *tmp;
}

汇编代码:

; Attributes: bp-based frame

; void __cdecl swap(int *a, int *b)
public __Z4swapPiS_
__Z4swapPiS_ proc near

var_18= qword ptr -18h
var_10= qword ptr -10h
tmp= qword ptr -8
b= qword ptr  0
a= qword ptr  8

push    rbp
mov     rbp, rsp
mov     [rbp+tmp], rdi
mov     [rbp+var_10], rsi
mov     rax, [rbp+tmp]
mov     ecx, [rax]
mov     rax, [rbp+var_18]
mov     [rax], ecx
mov     rax, [rbp+var_10]
mov     ecx, [rax]
mov     rax, [rbp+tmp]
mov     [rax], ecx
mov     rax, [rbp+var_18]
mov     ecx, [rax]
mov     rax, [rbp+var_10]
mov     [rax], ecx
pop     rbp
retn
__Z4swapPiS_ endp

汇编代码解释(分号表示注释)

注解

[] 表示读取该地址的值

; Attributes: bp-based frame

; void __cdecl swap(int *a, int *b)
public __Z4swapPiS_
__Z4swapPiS_ proc near

; 定义函数局部变量, 局部变量放在 负偏移 (相对于 rbp 函数基址的偏移量)
var_18= qword ptr -18h
var_10= qword ptr -10h
tmp= qword ptr -8

; 定义函数的参数, 参数在 正偏移
b= qword ptr  0
a= qword ptr  8

; 栈帧设置
;   当进入一个函数时，通常会将调用者的 rbp 值压入栈中（使用 push rbp），然后将 rsp （栈指针）赋值给 rbp。这样，rbp 就成为了当前函数的基指针。
; 目的：将调用者的基指针 rbp 压入栈中，以保存调用者的栈帧。
; 结果：栈顶指针 rsp 减小，栈中保存了调用者的 rbp 值。
push    rbp

; 目的：将当前栈指针 rsp 的值赋给 rbp，建立新的栈帧。
; 结果：rbp 现在指向当前函数的栈帧底部，允许通过相对偏移访问局部变量和参数。
mov     rbp, rsp

; 变量赋值
; 将传入的第一个参数（通常在 rdi 中）存储到 tmp 变量中。
;   *tmp = a
mov     [rbp+tmp], rdi
; 将传入的第二个参数（通常在 rsi 中）存储到 var_10 变量中。
;   *var_10 = b
mov     [rbp+var_10], rsi

;   *var_18 = *tmp
; 从 tmp 中读取地址（然后 rax 包含 tmp 存储的值）。
mov     rax, [rbp+tmp]
; 从 rax 指向的地址中读取 4 字节的数据到 ecx 中。
mov     ecx, [rax]
; 上面取了两次值
; 从 var_18 中读取地址（rax 现在包含 var_18 存储的值）。  放到 rax 寄存器
mov     rax, [rbp+var_18]
; 将 ecx 中的值写入 rax 指向的地址（即 var_18 所指向的内存）
mov     [rax], ecx
; 上面又取了两次值

; 重复
;   从 var_10 中读取值，再写入到 tmp 中, (tmp=var_10)
;     *tmp = *var_10;
mov     rax, [rbp+var_10]
mov     ecx, [rax]
mov     rax, [rbp+tmp]
mov     [rax], ecx

; 重复
;   从 var_18 中读取值，再写入到 var_10 中, (var_10=var_18)
;     *var_10=*var_18
mov     rax, [rbp+var_18]
mov     ecx, [rax]
mov     rax, [rbp+var_10]
mov     [rax], ecx

; 上面重复的内容合起来看就是
;   *var_18 = *tmp;
;   *tmp = *var_10;
;   *var_10=*var_18;
; 相当于源码的
;     *tmp = *a;
;     *a = *b;
;     *b = *tmp;
; 可以明显看出, 汇编中的变量和源码中的变量名不一定是一致, 比如源码的 `tmp` 这里是 `var_18`

; 恢复之前保存的基址指针
pop     rbp
; 返回到调用该函数的地方。
retn
__Z4swapPiS_ endp

减法

源码

int sub(int a, int b){
  return a-b;
}

汇编代码:

; int __cdecl sub(int a, int b)
public __Z3subii
__Z3subii proc near

; 局部变量 var_8, var_4
var_8= dword ptr -8
var_4= dword ptr -4

; 参数, 这里为啥只有一个不清楚
b= dword ptr  8

; 存储调用位置基址(函数结束时返回用), 然后将当前栈顶地址作为新的栈帧基址
push    rbp
mov     rbp, rsp

; *var_4 = a
mov     [rbp+var_4], edi
; *var_8 = b
mov     [rbp+var_8], esi
; eax = *var_4
mov     eax, [rbp+var_4]
; eax -= *var_8
; 即: eax = *var_4 - *var_8
sub     eax, [rbp+var_8]
pop     rbp
retn
__Z3subii endp
