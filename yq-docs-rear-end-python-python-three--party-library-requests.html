requests

修改于: 2023-02-20

部分参数

stream=False

是否开启流式传输, 默认关闭.

当开始流式传输时候, 不会立刻开始下载, 比如

stream_obj = requests.get(self.url, stream=True)
total_size = int(stream_obj.headers['Content-Length'])

stream_obj 仅创建了一个流式传输对象, 并没有开始实际的传输, 需要手动调用 iter_content(chunk_size=1024) 或者 iter_lines 来开启真正的传输

with open("data.xxx", 'wb') as f:
  for chunk in res_left.iter_content(chunk_size=1024):
    f.write(chunk)
    # f.flush()

其中 chunk_size 不应设置的过大, 否则可能会因为带宽不足导致 ConnectionResetError(10054, '远程主机强迫关闭了一个现有的连接。', None, 10054, None))

可以通过这个参数来支持断点续传, 使用方式是在 headers 指定传输的起始位置

# 不指定结束
# 或者 headers = {'Range': 'bytes=%d-'%(start,)}
# 指定结束
headers = {'Range': 'bytes=%d-%d'%(start,end)}

大致完整代码

import os, requests


def download_with_resume(
        url: str,
        save_file: str,
):

  start_size = 0
  if os.path.isfile(save_file):
      start_size = os.path.getsize(save_file)

  headers = {
      "Range": "bytes={}-".format(start_size),
  } if start_size else None

  ret_response = requests.get(url, stream=True, headers=headers)
  # 206 表示只请求部分资源
  if ret_response.status_code not in (200, 206):
      raise RuntimeError("Download failed ...")

  # 可能还得判断下 start_size 超过总大小情况

  with open(save_file, "ab") as f:
      for chunk in ret_response.iter_content(chunk_size=1024):
          if chunk:
              f.write(chunk)
              f.flush()

断点续传:

import sys
import requests
import os


class Downloader(object):
    def __init__(self, url, file_path):
        self.url = url
        self.file_path = file_path

    def start(self):
        res_length = requests.get(self.url, stream=True)
        total_size = int(res_length.headers['Content-Length'])
        print(res_length.headers)
        print(res_length)
        if os.path.exists(self.file_path):
            temp_size = os.path.getsize(self.file_path)
            print("当前：%d 字节， 总：%d 字节， 已下载：%2.2f%% " % (temp_size, total_size, 100 * temp_size / total_size))
        else:
            temp_size = 0
            print("总：%d 字节，开始下载..." % (total_size,))

        headers = {'Range': 'bytes=%d-' % temp_size,
                  "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64; rv:81.0) Gecko/20100101 Firefox/81.0"}
        res_left = requests.get(self.url, stream=True, headers=headers)

        with open(self.file_path, "ab") as f:
            for chunk in res_left.iter_content(chunk_size=1024):
                temp_size += len(chunk)
                f.write(chunk)
                f.flush()

                done = int(50 * temp_size / total_size)
                sys.stdout.write("\r[%s%s] %d%%" % ('█' * done, ' ' * (50 - done), 100 * temp_size / total_size))
                sys.stdout.flush()


if __name__ == '__main__':
    url = "http://www.test.com/xxxxx/test.jpg"
    path = r"c:\test.jpg"
    downloader = Downloader(url, path)
    downloader.start()

注解

若文件下载 url 存在重定向, 则使用 allow_redirects=True

requests.get(url, allow_redirects=True)

分块下载使用 stream=True

requests.get(url, stream=True)

断点续传需要指定 headers 下的 Range 字段 来指定范围

# 或者 headers = {'Range': 'bytes=%d-'%(start,)}
headers = {'Range': 'bytes=%d-%d'%(start,end)}

关于文件下载时候的路径

当从一个url下载文件时候, 若使用绝对路径, 不会有问题.

但可能有时候不想使用绝对路径, 而是使用相对路径, 那么这个时候可能就会 存在给定的下载路径正确, 但就是说路径不存在, 下载不了, 也没发使用python执行(若是可执行文件).

这个时候可能是文件名的命名使用了url编码的缘故, 而其中可能有些特殊字符不能被python正常解析, 故而导致的这个问题 将名称使用url解码即可:

urllib.parse.unquote(url_path)

附-断点续传代码

import hashlib
import logging
import os, requests
from typing import Union

_LOGGER = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO)


def read_file(file: str, mode="r", size=-1) -> Union[bytes, str]:
    with open(file, mode) as f:
        return f.read(size)


def read_file_one_by_one(file: str, mode="rb", size=1024):
    with open(file, mode) as f:
        yield f.read(size)


# 检查前部分内容是否相同, 相同才允许继续下载
def check_file_right(
        url: str,
        save_file: str,
):
    if os.path.isfile(save_file):
        compare_size = 256
        head_bytes: bytes = read_file(save_file, 'rb', compare_size)
        ret_response = requests.get(url, stream=False, headers={
            # 高版本减去一个, 可以理解为 bytes 的范围是索引, 两端都包含
            "Range": "bytes=0-{}".format(compare_size-1),
        })
        return head_bytes == ret_response.content
    return True


# 检查是不是已经全部下载, 通过hash计算
def check_file_sha1(
        check_file: str,
        hope_sha1: str,
):

    sha1_ = hashlib.sha1()
    if os.path.isfile(check_file):
        for dat in read_file_one_by_one(check_file):
            sha1_.update(dat)
        # hashlib.new("sha1", all_bytes)
        return sha1_.hexdigest() == hope_sha1

    return False


# 检查是不是已经全部下载, 如果服务端有提供 hash 值, 可以通过计算hash值来判断
def check_is_all_file(
        url: str,
        save_file: str,
        file_size: int,
):
    if file_size and os.path.isfile(save_file):
        ret_response = requests.get(url, stream=True)
        all_size: str = ret_response.headers.get('content-length')
        return str(all_size) == str(file_size)
    return False


def download_with_resume(
        url: str,
        save_file: str,
):

    start_size = 0
    if os.path.isfile(save_file):
        start_size = os.path.getsize(save_file)
        if not check_file_right(url, save_file):
            _LOGGER.info("check local file not right, reDownload...")
            start_size = 0
        else:
            _LOGGER.info("local file downloaded some ... resume")

    # 检查是不是已经下载完成
    if check_is_all_file(url, save_file, start_size):
        _LOGGER.info("local file already downloaded, return")
        return

    headers = {
        "Range": "bytes={}-".format(start_size),
    } if start_size else None

    ret_response = requests.get(url, stream=True, headers=headers)
    # 206 表示只请求部分资源
    if ret_response.status_code not in (200, 206):
        raise RuntimeError("Request resources failed ...")

    # 可能还得判断下 start_size 超过总大小情况

    _LOGGER.info("Download start, wait...")
    with open(save_file, "ab" if start_size else "wb") as f:
        for chunk in ret_response.iter_content(chunk_size=1024):
            if chunk:
                f.write(chunk)
                f.flush()
    _LOGGER.info("Download end")


def test_download():
    download_with_resume(
        "https://www.kumeiwp.com/wj/196423/2024/02/04/4899cc8c109ec3cf6ef9c9e237d32d16.mp3?xz=b1",
        "吹梦到西洲.mp3"
    )


if __name__ == "__main__":
    test_download()
