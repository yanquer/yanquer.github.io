getopt.h

修改于: 2024-08-12

此库的作用就是命令行参数处理, 包括

  • 长选项
  • 短选项
  • 位置参数

函数原型-常用

#include <getopt.h>

// 短选项函数
int getopt(int argc, char * const argv[], const char *optstring);

// 长选项函数
int getopt_long(int argc, char * const argv[],const char *optstring, const struct option *longopts,int *longindex);

option结构体

原型

struct option {
  const char *name;     // 长选项名称
  int has_arg;          // 是否有参数值
  int *flag;            // 如果不为NULL,保存选项值到该变量
  int val;              // 如果flag为NULL,返回该值
};

各个成员变量的含义如下:

name
长选项的名称,以字符串形式表示。例如"help"或"version"。
has_arg

表示该选项是否需要参数值。取值有三种:

  • no_argument (0): 表示该选项不需要参数值。
  • required_argument (1): 表示该选项需要一个参数值。
  • optional_argument (2): 表示该选项的参数值是可选的。
flag
如果不为 NULL，则 getopt_long() 函数会将选项的值保存到该变量指向的位置,而不是返回该值。
val
如果 flag 变量为 NULL，则 getopt_long() 函数会返回该值作为 短选项标识符。

使用 option 结构体可以方便地定义命令行选项,并通过 getopt_long() 函数解析它们。例如:

static struct option long_options[] = {
    {"help", no_argument, NULL, 'h'},
    {"version", no_argument, NULL, 'v'},
    {0, 0, 0, 0}
};

这里定义了两个长选项: "help" 和 "version" , 它们分别对应短选项 -h 和 -v 。当解析到这两个选项时, getopt_long() 函数会返回 'h' 和 'v' 作为选项标识符。

使用实例-长选项函数

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <getopt.h>

typedef struct {
    char *launch_dir;
    BOOL launch_dir_need_free;
    char *app_name;
    BOOL app_name_need_free;
    // 用于启动 app 时使用的参数
    char *app_args[20];
} LauncherArgs;

const char *USAGE = "Usage: \n"
                    "[-h|--help]                    help message\n"
                    "[-v|--version]                 version info\n"
                    "[-d|--launch_dir <dir>]        the dir of versions_dir's install\n"
                    "[-n|--app_name <name>]         the exe-name in version_dir\n"
                    "\n"
                    "if has args for <app_name>, use -- \n"
                    "\n";

static struct option LongOptions[] = {
        {"help",    no_argument,       0, 'h'},
        {"version", no_argument,       0, 'v'},
        {"launch_dir", optional_argument,       0, 'd'},
        {"app_name",  required_argument, 0, 'n'},
        {0, 0, 0, 0}
};


void print_help_and_exit(){
    printf("%s", USAGE);
    exit(0);
}

void set_default_args(LauncherArgs *args, char *start_path){
    char cwd[1024];
    get_cur_path(cwd);

    if (args->launch_dir == NULL){
        args->launch_dir_need_free = TRUE;
        args->launch_dir = malloc(1024);
        strcpy(args->launch_dir, cwd);
    }

    if (args->app_name == NULL && start_path != NULL){
        args->app_name_need_free = TRUE;
        args->app_name = malloc(1024);
        get_basename(start_path, args->app_name);
    }

    info("set_default_args end");

}


void parse_cli_args(int argc, char *argv[], LauncherArgs *args) {
    int c;
    int option_index = 0;

    int should_end = 0;
    while ((c = getopt_long(argc, argv, "hvn:d::",
                            LongOptions, &option_index)) != -1) {
        switch (c) {
            case 'h':
                print_help_and_exit();
                info("parse_cli_args with help ... ");
                break;
            case 'v':
                printf("%s", "0.1.0");
                print_help_and_exit();
                break;
            case 'd':
                args -> launch_dir = optarg;
                break;
            case 'n':
                args -> app_name = optarg;
                break;
//            case 0:
//                opt_end = 1;
//                break;
            default:
                // 不检查不符合要求的参数
                should_end = 1;
                // info("parse_cli_args un-support args ... ");
                // print_help_and_exit();
                break;
        }
        if (should_end) break;
    }

    char msg[1024];
    int cur = 0;

    // 非选项参数
    for (int i=optind > 0 ? optind-1 : optind; i < argc; i++) {
        if (strcmp(argv[i], "--") == 0 || i==0){
            continue;
        }
        sprintf(msg, "Non-option argument: %s", argv[i]);
        info(msg);
        args->app_args[cur++] = argv[i];
    }

    if (args->app_name == NULL || args->launch_dir == NULL){
        set_default_args(args, argv[0]);
    }

}
